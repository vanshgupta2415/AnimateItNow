            function copyCode(codeBlock, btn) {
                btn.style.display = "none";
                const code = codeBlock.innerText;
                btn.style.display = "";
                navigator.clipboard.writeText(code).then(() => {
                  btn.textContent = "Copied!";
                  setTimeout(() => {
                    btn.textContent = "Copy";
                  }, 1200);
                });
              }
